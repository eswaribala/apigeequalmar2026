kubectl annotate storageclass gp2 storageclass.kubernetes.io/is-default-class="true" --overwrite
kubectl get csidrivers
aws eks list-clusters
aws eks create-addon --cluster-name apigee-hybrid-eks --addon-name aws-ebs-csi-driver
kubectl get pods -n kube-system | egrep 'ebs|csi'

cat <<'EOF' | kubectl apply -f -
apiVersion: storage.k8s.io/v1
kind: StorageClass
metadata:
  name: gp3-csi
  annotations:
    storageclass.kubernetes.io/is-default-class: "true"
provisioner: ebs.csi.aws.com
volumeBindingMode: WaitForFirstConsumer
allowVolumeExpansion: true
parameters:
  type: gp3
  fsType: ext4
EOF
